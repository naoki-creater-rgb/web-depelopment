-- 1. データベースの作成
CREATE DATABASE IF NOT EXISTS event_app_db CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE event_app_db;

-- 2. users テーブル
CREATE TABLE users (
    user_id VARCHAR(255) PRIMARY KEY,
    password_hash VARCHAR(255) NOT NULL,
    display_name VARCHAR(255) NOT NULL
);

-- 3. events テーブル
CREATE TABLE events (
    event_id INT AUTO_INCREMENT PRIMARY KEY,
    manager_id VARCHAR(255) NOT NULL,
    event_name VARCHAR(255) NOT NULL,
    response_deadline DATETIME NOT NULL,
    description TEXT,
    status VARCHAR(50) DEFAULT 'unconfirmed',
    confirmed_area_id VARCHAR(255),
    confirmed_shop_name VARCHAR(255),
    confirmed_budget VARCHAR(255),
    confirmed_datetime_id DATETIME,
    payment_destination VARCHAR(255),
    paypay_link VARCHAR(255),
    FOREIGN KEY (manager_id) REFERENCES users(user_id) ON DELETE CASCADE
);



-- 5. event_date_candidates テーブル (日程候補)
CREATE TABLE event_date_candidates (
    date_candidate_id INT AUTO_INCREMENT PRIMARY KEY,
    event_id INT NOT NULL,
    proposed_datetime DATETIME NOT NULL,
    total_score INT DEFAULT 0,
    FOREIGN KEY (event_id) REFERENCES events(event_id) ON DELETE CASCADE
);

-- 6. event_area_candidates テーブル (エリア候補)
CREATE TABLE event_area_candidates (
    area_candidate_id INT AUTO_INCREMENT PRIMARY KEY,
    event_id INT NOT NULL,
    proposed_area VARCHAR(255) NOT NULL,
    total_score INT DEFAULT 0,
    FOREIGN KEY (event_id) REFERENCES events(event_id) ON DELETE CASCADE
);

-- 4. event_participants テーブル (参加者と基本回答)
CREATE TABLE event_participants (
    event_id INT NOT NULL,
    user_id VARCHAR(255) NOT NULL,
    preferred_budget INT,
    preferred_area_id INT,
    overall_comment TEXT,
    PRIMARY KEY (event_id, user_id),
    FOREIGN KEY (event_id) REFERENCES events(event_id) ON DELETE CASCADE,
    FOREIGN KEY (user_id) REFERENCES users(user_id) ON DELETE CASCADE,
    FOREIGN KEY (preferred_area_id) REFERENCES event_area_candidates(area_candidate_id) ON DELETE CASCADE
);

-- 7. participant_date_responses テーブル (日程別の回答)
CREATE TABLE participant_date_responses (
    event_id INT NOT NULL,
    user_id VARCHAR(255) NOT NULL,
    date_candidate_id INT NOT NULL,
    score VARCHAR(50) NOT NULL,
    comment TEXT,
    PRIMARY KEY (event_id, user_id, date_candidate_id),
    FOREIGN KEY (event_id, user_id) REFERENCES event_participants(event_id, user_id) ON DELETE CASCADE,
    FOREIGN KEY (date_candidate_id) REFERENCES event_date_candidates(date_candidate_id) ON DELETE CASCADE
);
